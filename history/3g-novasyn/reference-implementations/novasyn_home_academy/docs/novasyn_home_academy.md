      `${subject.name}_${subject.grade_level}_${Date.now()}.nsacademy`
    )

    return filePath

    await db.run(`

### **How Accommodations Affect Generated Content**

```javascript
class AccommodationEngine {
  
  async applyAccommodations(studentId, generatedContent, contentType) {
    const accommodations = await db.all(`
      SELECT * FROM academy_accommodations
      WHERE student_id = ? AND is_active = 1
    `, [studentId])
    
    if (accommodations.length === 0) return generatedContent
    
    let modified = { ...generatedContent }
    
    for (const accommodation of accommodations) {
      const profile = JSON.parse(accommodation.profile_settings)
      
      switch (accommodation.profile_type) {
        
        case 'adhd':
          modified = await this.applyADHD(modified, profile, contentType)
          break
          
        case 'dyslexia':
          modified = await this.applyDyslexia(modified, profile, contentType)
          break
          
        case 'gifted':
          modified = await this.applyGifted(modified, profile, contentType)
          break
          
        case 'autism':
          modified = await this.applyAutism(modified, profile, contentType)
          break
          
        case 'dyscalculia':
          modified = await this.applyDyscalculia(modified, profile, contentType)
          break
          
        case 'dysgraphia':
          modified = await this.applyDysgraphia(modified, profile, contentType)
          break
          
        case 'custom':
          modified = await this.applyCustom(modified, profile, contentType)
          break
      }
    }
    
    return modified
  }
  
  async applyADHD(content, profile, contentType) {
    if (contentType === 'lesson_plan') {
      // Break lesson into shorter segments
      const maxSegmentMinutes = profile.max_segment_minutes || 12
      
      // Add movement breaks
      content.movement_breaks = this.insertMovementBreaks(
        content, maxSegmentMinutes
      )
      
      // Increase estimated time by 25% (transitions take longer)
      content.estimated_minutes = Math.ceil(
        content.estimated_minutes * 1.25
      )
      
      // Add fidget suggestions
      content.accommodations_notes = 
        `ADHD accommodations applied:
         - Lesson broken into ${maxSegmentMinutes}-min segments
         - Movement break every ${maxSegmentMinutes} minutes
         - Consider allowing fidget tools during seated work
         - If focus drops, switch to the hands-on activity early
         - End on a high note — don't push past attention limit`
    }
    
    if (contentType === 'worksheet') {
      // Fewer items per page
      content.items_per_page = Math.min(
        content.items_per_page, 
        profile.max_items_per_page || 5
      )
      
      // More white space
      content.layout = 'spacious'
      
      // Add visual breaks between problems
      content.visual_separators = true
    }
    
    if (contentType === 'assessment') {
      // Break into sections with breaks between
      content.sections = content.sections.map(section => ({
        ...section,
        break_after: true,
        break_activity: 'Stand up, stretch, get water'
      }))
      
      // Extend time estimate
      content.estimated_minutes = Math.ceil(
        content.estimated_minutes * 1.5
      )
    }
    
    return content
  }
  
  async applyDyslexia(content, profile, contentType) {
    if (contentType === 'worksheet' || contentType === 'assessment') {
      // Font and sizing adjustments
      content.font = profile.preferred_font || 'OpenDyslexic'
```
